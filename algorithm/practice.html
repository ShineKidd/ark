function getMaxSubStr (str1, str2) {
  let maxStr, subStr
  if (str1 > str2) {
    maxStr = str1
    subStr = str2
  } else {
    maxStr = str2
    subStr = str1
  }

  let subStrLen = subStr.length
  for (let i = 0; i < subStrLen; i++) {
    // ç§»åŠ¨å¤´å°¾æŒ‡é’ˆ
    let start = 0, end = subStrLen - i
    while(end <= subStrLen) {
      const currStr = subStr.slice(start, end)
      if (maxStr.includes(currStr)) {
        return currStr
      }
      start++
      end++
    }
  }
}
