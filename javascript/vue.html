const LoadingView = {
  functional: true,
  props: {},
  render (h, context) {
    return h('div', {}, 'Loading...')
  },
}

const ErrorView = {
  functional: true,
  props: {},
  render (h, context) {
    return h('div', {}, 'Unexception Error')
  },
}

function createLazy (component) {
  const View = () => ({
    component,
    loading: LoadingView,
    error: ErrorView,
    delay: 200,
    timeout: 3000
  })

  return Promise.resolve({
    functional: true,
    render (h, { data, children }) => h(View, data, chilren),
  })
}

const routes = [
  {
    path: '/user',
    component: () => createLazy(import('@/views/user/index.vue')),
  },
]
